AIRPORT TABLE
===================

create table Airport(AirportId int NOT NULL,
AirportName varchar(255) NOT NULL,
FOREIGN KEY (CityId) REFERENCES City(CityId),
PRIMARY KEY (AirportId),
FOREIGN KEY (StateId) REFERENCES State(Stateid));

HUBS TABLE
==============

create table Hubs(HubId SMALLINT UNSIGNED NOT NULL,
HubName varchar(100) NOT NULL,
ContactNo bigint unsigned NOT NULL,
Address varchar(255) NOT NULL,
PRIMARY KEY (HubId),
FOREIGN KEY (CityId) REFERENCES City(CityId),FOREIGN KEY (StateId) REFERENCES State(StateId));
